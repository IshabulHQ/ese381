117 // DESCRIPTION
118 // Configures the SAML21J18B's SERCOM1 for SPI communication with the
119 // BME680. PA16 = MOSI, PA17 = SCK, PA19 = MISO, PB07 = CS.
120 //
121 // Warnings : none
122 // Restrictions : none
123 // Algorithms : none
124 // References : none
125 //
126 // Revision History : Initial version
127 //
128 //**************************************************************************
129 static void init_spi_BME680 (void) {
130  REG_GCLK_PCHCTRL19 = 0x00000040; /* SERCOM1 core clock not enabled by default */
131  ARRAY_PORT_PINCFG0[16] |= 1; /* allow pmux to set PA16 pin configuration */
132  ARRAY_PORT_PINCFG0[17] |= 1; /* allow pmux to set PA17 pin configuration */
133  ARRAY_PORT_PINCFG0[18] |= 1; /* allow pmux to set PA18 pin configuration */
134  ARRAY_PORT_PINCFG0[19] |= 1; /* allow pmux to set PA19 pin configuration */
135  ARRAY_PORT_PMUX0[8] = 0x22; /* PA16 = MOSI, PA17 = SCK */
136  ARRAY_PORT_PMUX0[9] = 0x22; /* PA18 = SS, PA19 = MISO */
137  REG_PORT_DIRSET1 = 0x80; /* PB07 = CS for BME680 */
138  REG_PORT_DIRCLR0 = 0x04; //PA02 input for SW0
139  ARRAY_PORT_PINCFG0[2] |= 6; //enable PA02 with pull
140  REG_PORT_OUTSET0 = 0x04; //make PA02 pull-up
141  REG_SERCOM1_SPI_CTRLA = 1; /* reset SERCOM1 */
142  while (REG_SERCOM1_SPI_CTRLA & 1) {} /* wait for reset to complete */
143  REG_SERCOM1_SPI_CTRLA = 0x3030000C; /* MISO-3, MOSI-0, SCK-1, SS-2, CPOL=1, CPHA=1 */
144  REG_SERCOM1_SPI_CTRLB = 0x00020000; /* Master SS, 8-bit, receiver enabled */
145  REG_SERCOM1_SPI_BAUD = 0; /* SPI clock is 4MHz/2 = 2MzHz */
146  REG_SERCOM1_SPI_CTRLA |= 2; /* enable SERCOM1 */
147 }

229 static uint8_t spi_transfer (uint8_t data) {
230  uint8_t Rx_data;
231  while(!(REG_SERCOM1_SPI_INTFLAG & 1)) {} //wait until Tx ready
232  REG_SERCOM1_SPI_DATA = data; //send data byte
233  while(!(REG_SERCOM1_SPI_INTFLAG & 2)) {} //wait until transmit is complete
234  while(!(REG_SERCOM1_SPI_INTFLAG & 4)) {} //wait until receive is complete
235  Rx_data = REG_SERCOM1_SPI_DATA; //read data register
236  return Rx_data;
237 }

245 // Author : Brandon Cheung, Ishabul Haque
246 // DESCRIPTION
247 // This function is passed 4 parameters, only reg_addr, *reg_data and len are
248 // used in this definition. The MS bit of reg_addr is masked to a '1' in order
249 // to indicate a read transaction to the BME680. The spi_transfer function
250 // is first called to pass the register address to be read. It is then called
251 // a second time to receive the register's data and update *reg_data. If
252 // len > 1, the register address to be read is auto-incremented and the address
253 // of reg_data is incremented as well to allow for multiple read transactions.
254 //
255 // Warnings : none
256 // Restrictions : none
257 // Algorithms : none
258 // References : none
259 //
260 // Revision History : Initial version
261 //
262 //**************************************************************************
263 int8_t user_spi_read (uint8_t dev_id, uint8_t reg_addr, uint8_t *reg_data, uint16_t len) {
264  int8_t rslt = 0; //return 0 for success, non-zero for failure
265  reg_addr |= 0x80; //mask bit 7 to a '1' for a read transaction
266  REG_PORT_OUTCLR1 &= 0x80; //CS = 0 -> BME680 selected
267  spi_transfer(reg_addr); //send control byte with register address
268  while (len--) {
269  *reg_data = spi_transfer(0);
270  reg_data++;
271  }
272  REG_PORT_OUTSET1 |= 0x80; //CS = 1 -> BME680 unselected
273  return rslt;
274 }
