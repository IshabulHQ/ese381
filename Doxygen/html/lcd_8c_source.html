95 extern void init_spi_lcd (void) {
96  REG_GCLK_PCHCTRL19 = 0x00000040; /* SERCOM1 core clock not enabled by default */
97 
98  ARRAY_PORT_PINCFG0[16] |= 1; /* allow pmux to set PA16 pin configuration */
99  ARRAY_PORT_PINCFG0[17] |= 1; /* allow pmux to set PA17 pin configuration */
100  ARRAY_PORT_PINCFG0[18] |= 1; /* allow pmux to set PA18 pin configuration */
101  ARRAY_PORT_PINCFG0[19] |= 1; /* allow pmux to set PB06 pin configuration */
102  ARRAY_PORT_PMUX0[8] = 0x22; /* PA16 = MOSI, PA17 = SCK */
103  ARRAY_PORT_PMUX0[9] = 0x22; /* PA18 = SS, PA19 = MISO */
104 
105  REG_PORT_DIRSET1 = 0x40; /* RS output */
106 
107  REG_SERCOM1_SPI_CTRLA = 1; /* reset SERCOM1 */
108  while (REG_SERCOM1_SPI_CTRLA & 1) {} /* wait for reset to complete */
109 
110  REG_SERCOM1_SPI_CTRLA = 0x3030000C; /* MISO-3, MOSI-0, SCK-1, SS-2, CPOL=1, CPHA=1 */
111  REG_SERCOM1_SPI_CTRLB = 0x00002000; /* Master SS, 8-bit */
112  REG_SERCOM1_SPI_BAUD = 0; /* SPI clock is 4MHz/2 = 2MzHz */
113  REG_SERCOM1_SPI_CTRLA |= 2; /* enable SERCOM1 */
114 }

137 extern void lcd_spi_transmit_CMD (char CMD) {
138  REG_PORT_OUTCLR1 = 0x00000040; // RS = 0 -> command, /SS = 0 -> selected
139  while(!(REG_SERCOM1_SPI_INTFLAG & 1)) {} /* wait until Tx ready */
